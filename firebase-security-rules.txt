// ==============================
// FIREBASE SECURITY RULES
// ==============================

// FIRESTORE DATABASE RULES
// Copy these rules to: Firebase Console → Firestore → Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Events Collection
    match /events/{eventId} {
      // Anyone can read events (public access)
      allow read: if true;
      
      // Only authenticated admin users can create, update, or delete events
      allow create, update, delete: if request.auth != null;
    }
    
    // Add more collections here as needed
  }
}

// ==============================
// FIREBASE STORAGE RULES
// ==============================

// STORAGE RULES
// Copy these rules to: Firebase Console → Storage → Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // Events images folder
    match /events/{imageId} {
      // Anyone can read/view images (public access)
      allow read: if true;
      
      // Only authenticated users can upload images
      allow write: if request.auth != null 
                   && request.resource.size < 5 * 1024 * 1024  // 5MB max file size
                   && request.resource.contentType.matches('image/.*');  // Only images
    }
    
    // Add more storage paths here as needed
  }
}

// ==============================
// HOW TO APPLY THESE RULES
// ==============================

/*
STEP 1: Firestore Rules
1. Go to Firebase Console: https://console.firebase.google.com/
2. Select your project
3. Go to Firestore Database → Rules tab
4. Copy the FIRESTORE DATABASE RULES above
5. Paste them into the rules editor
6. Click "Publish"

STEP 2: Storage Rules
1. In Firebase Console
2. Go to Storage → Rules tab
3. Copy the STORAGE RULES above
4. Paste them into the rules editor
5. Click "Publish"

IMPORTANT NOTES:
- These rules allow anyone to READ events and images (public access)
- Only authenticated admin users can CREATE/UPDATE/DELETE events
- Only authenticated users can UPLOAD images
- Images are limited to 5MB file size
- Only image file types are allowed

SECURITY TIPS:
- Never share your admin credentials
- Use strong passwords for admin accounts
- Regularly review Firebase Authentication users
- Monitor Firebase Usage to detect unusual activity
- Enable Firebase App Check for production (optional)
- Consider adding more specific rules based on user roles
*/
